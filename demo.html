ScoreBoard style panel view

How to use

Time

time HH:mm #scoreboard-demo-2

$("#scoreboard-demo-2").sbPanel({
  format: 'HH:mm',
  value: '11:33'
});

time HH:mm:ss #scoreboard-demo-3

$("#scoreboard-demo-3").sbPanel();

clock #scoreboard-demo-4

$("#scoreboard-demo-4").sbPanel({
  format : 'HH:mm:ss',
  value : new Date().toLocaleTimeString()
});
var t = window.setInterval(function( ) {
  $("#scoreboard-demo-4").sbPanel('update',
      { value: new Date().toLocaleTimeString() });
}, 1000);

Date

date DD/MM/YYYY #scoreboard-demo-5

$("#scoreboard-demo-5").sbPanel({
  format : 'dd/mm/yyyy',
  value : '19/07/2015'
});

date DD-MM-YYYY #scoreboard-demo-6

$("#scoreboard-demo-6").sbPanel({
  format : 'dd-mm-yyyy',
  value : new Date().toLocaleDateString()
});

#scoreboard-demo-7

$("#scoreboard-demo-7").sbPanel({
  format : 'dd.mm.yyyy',
  value : new Date('1945-05-09').toLocaleDateString()
});

#scoreboard-demo-8 with external control field

$("#scoreboard-demo-8").sbPanel({
	format: 'YYYY-MM-DD',
	control: $("#myControlField")
});
